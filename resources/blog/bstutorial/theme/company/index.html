<!-- Use jQuery to add smooth scrolling (when clicking on the links in the navbar) -->
<script>
$(document).ready(function() {
  // Add smooth scrolling to all links in navbar + footer links
  $(".navbar a, footer a[href='#myPage']").on('click', function(event) {

    // Make sure this.hash has a value before overriding default behavior
    if(this.hash !== "") {
      // prevent default anchor click behavior
      event.preventDefault();

      // store hash
      var hash = this.hash;

      // Using jQuery's animate() method to add smooth page scroll
      // The optional number (400) specifies the number of milliseconds it takes to scroll to the specified area
      $('html, body').animate({
        scrollTop: $(hash).offset().top
      }, 400, function() {

      });
    }   // end if
  });

  /*
  We have also created an animation effect which will slide in elements on scroll.
  If you want to use it, just add the .slideanim class to the element you want to slide in,
  and add the following to your CSS and jQuery (feel free to modify the duration, opacity,
  where to start, when to slide in, and so on):
  */

  $(window).scroll(function() {
    $(".slideanim").each(function() {
      var pos = $(this).offset().top;

      var winTop = $(window).scrollTop();
      if(pos < winTop + 600) {
        $(this).addClass("slide");
      }
    });
  });

});
</script>
